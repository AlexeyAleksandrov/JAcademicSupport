% ── ЛЕВАЯ КОЛОНКА ─────────────────────────────────────────────────────────────
\begin{column}{0.50\textwidth}
\begin{block}{\bfseries Параметры источника $s \in \mathcal{S}$}
\small
\begin{itemize}
  \item Входной кортеж: $(n^{\mathrm{tot}}_s,\;n^{\mathrm{rel}}_s,\;\bar{a}_s,\;\alpha_s,\;w_s,\;\lambda_s)$
  \item Эффективный объём:
        $n^{\mathrm{eff}}_s = n^{\mathrm{rel}}_s \cdot \alpha_s$
  \item Коэффициент уверенности:
        \[
          \kappa_s = \dfrac{n^{\mathrm{eff}}_s}{n^{\mathrm{eff}}_s + \lambda_s}
          \;\in (0,1)
        \]
  \item BPA для vac, exp:
        \[
          m_s(\{T\}) = \kappa_s \cdot \bar{a}_s \cdot \frac{n^{\mathrm{rel}}_s}{n^{\mathrm{tot}}_s},
          \quad m_s(\{F\}) = 0
        \]
  \item BPA для смешанных прогнозов fc:
        \[
          m_{\mathrm{fc}}(\{F\}) = \kappa_{\mathrm{fc}} \cdot \bar{a}_- \cdot \frac{n_-}{n^{\mathrm{tot}}_{\mathrm{fc}}}
        \]
  \item $m_s(\Theta) = 1 - m_s(\{T\}) - m_s(\{F\})$
\end{itemize}
\end{block}
\end{column}

% ── ПРАВАЯ КОЛОНКА ────────────────────────────────────────────────────────────
\begin{column}{0.48\textwidth}
\smallskip
\textbf{\color{lightgold}Дисконтирование по надёжности $w_s$:}
\[
  m'_s(A) = w_s \cdot m_s(A),\quad A \neq \Theta
\]
\[
  m'_s(\Theta) = (1 - w_s) + w_s \cdot m_s(\Theta)
\]

\smallskip
\begin{tabular}{lcc}
  \hline\rule{0pt}{2.2ex}
  Источник & $w_s$ & $\lambda_s$ \\
  \hline
  Эксперты & $0{,}90$ & $5$ \\
  Вакансии & $0{,}80$ & $15$ \\
  Прогнозы & $0{,}60$ & $2$ \\
  \hline
\end{tabular}

\bigskip
\textbf{\color{lightgold}Рамка рассуждений:}
\[
  \Theta = \{T,\,F\},\qquad
  2^\Theta = \bigl\{\emptyset,\{T\},\{F\},\Theta\bigr\}
\]

\smallskip
\textit{\footnotesize
Выход: дисконтированные тройки $(m'_s(\{T\}), m'_s(\Theta), m'_s(\{F\}))$
для каждого источника. Передаются на комбинирование по правилу Демпстера.}
\end{column}

% ── ПРАВАЯ КОЛОНКА — формулы ─────────────────────────────────────────────────
\begin{column}{0.60\textwidth}
\small

\textbf{\color{lightgold}Конфликт и правило Демпстера:}
\begin{align*}
  K &= m_1(\{T\})\cdot m_2(\{F\}) + m_1(\{F\})\cdot m_2(\{T\}) \\[4pt]
  (m_1{\oplus}m_2)(A) &=
    \frac{\displaystyle\sum_{B\cap C=A} m_1(B)\cdot m_2(C)}{1-K},
    \quad K < \tau_K \\[4pt]
  m^* &= m'_{\mathrm{vac}} \oplus m'_{\mathrm{exp}} \oplus m'_{\mathrm{fc}}
\end{align*}

\smallskip
\textbf{\color{lightgold}Пигнистическая вероятность и дефицит:}
\[
  \mathrm{BetP} = m^*\!\bigl(\{T\}\bigr) + \frac{m^*(\Theta)}{N_C},
  \qquad
  \Delta = \mathrm{BetP} - \mathrm{supply}
\]

\smallskip
\textbf{\color{lightgold}Правила принятия решений:}
\begin{align*}
  \Delta > \tau_\Delta \wedge K \le \tau_K \wedge m^*(\Theta) \le \tau_\Theta
    &\;\Rightarrow\; \textbf{УСИЛИТЬ} \\
  \Delta > \tau_\Delta \wedge \bigl(K > \tau_K \vee m^*(\Theta) > \tau_\Theta\bigr)
    &\;\Rightarrow\; \textbf{ВАРИАТИВНОСТЬ} \\
  |\Delta| \le \tau_\Delta
    &\;\Rightarrow\; \textbf{СТАБИЛИЗИРОВАТЬ}
\end{align*}

\vspace*{-0.4em}
{\footnotesize
$\tau_\Delta = 0{,}15$;\quad
$\tau_K = 0{,}40$;\quad
$\tau_\Theta = 0{,}15$;\quad
$N_C = 25$ (число кластеров программы).}
\end{column}
